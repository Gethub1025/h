def caesar_encrypt(text, shift):
result = ""
for char in text:
if char.isalpha():
base = ord('A') if char.isupper() else ord('a')
result += chr((ord(char) - base + shift) % 26 + base)
else:
result += char
return result
def caesar_decrypt(cipher, shift):
return caesar_encrypt(cipher, -shift)

2. encryption and decryption using a substitution cipher Python

<
def generate_key_map(key):
alphabet = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
key = key.upper()
return dict(zip(alphabet, key))
def invert_key_map(key_map):
return {v: k for k, v in key_map.items()}
def encrypt(text, key_map):
result = ""
for char in text:
if char.isalpha():
base = 'A' if char.isupper() else 'a'
mapped = key_map[char.upper()]
result += mapped if char.isupper() else mapped.lower()
else:
result += char
return result
def decrypt(cipher, key_map):
inverse_map = invert_key_map(key_map)
return encrypt(cipher, inverse_map)
# Example usage
key = "QWERTYUIOPASDFGHJKLZXCVBNM" # 26-letter substitution key
text = "Hello, People!"
key_map = generate_key_map(key)
encrypted = encrypt(text, key_map)
decrypted = decrypt(encrypted, key_map)
print("Original Text:", text)
print("Encrypted Text:", encrypted)
print("Decrypted Text:", decrypted)

3. Hill Cipher encryption and decryption Python


import numpy as np
# Convert text to numbers (A=0, B=1, ..., Z=25)
def text_to_numbers(text):
return [ord(char.upper()) - ord('A') for char in text if char.isalpha()]
# Convert numbers back to text
def numbers_to_text(numbers):
return ''.join([chr(num % 26 + ord('A')) for num in numbers])
# Pad text to match matrix size
def pad_text(text, size):
while len(text) % size != 0:
text += 'X' # Padding with 'X'
return text
# Encryption: C = (K × P) mod 26
def hill_encrypt(plain_text, key_matrix):
size = key_matrix.shape[0]
plain_text = pad_text(plain_text, size)
numbers = text_to_numbers(plain_text)
cipher = []
for i in range(0, len(numbers), size):
block = np.array(numbers[i:i+size])
encrypted_block = np.dot(key_matrix, block) % 26
cipher.extend(encrypted_block)
return numbers_to_text(cipher)
# Decryption: P = (K⁻¹ × C) mod 26
def hill_decrypt(cipher_text, key_matrix):
size = key_matrix.shape[0]
numbers = text_to_numbers(cipher_text)
# Compute inverse of key matrix mod 26
det = int(np.round(np.linalg.det(key_matrix)))
det_inv = pow(det, -1, 26) # Modular inverse of determinant
adj = np.round(det * np.linalg.inv(key_matrix)).astype(int) % 26
inverse_matrix = (det_inv * adj) % 26
plain = []
for i in range(0, len(numbers), size):
block = np.array(numbers[i:i+size])
decrypted_block = np.dot(inverse_matrix, block) % 26
plain.extend(decrypted_block)
return numbers_to_text(plain)
# Example usage
key = np.array([[3, 3],
[2, 5]]) # Must be invertible mod 26
text = "HELLO"
encrypted = hill_encrypt(text, key)
decrypted = hill_decrypt(encrypted, key)
print("Original Text:", text)
print("Encrypted Text:", encrypted)
print("Decrypted Text:", decrypted)

from Crypto.Cipher import DES
from Crypto.Util.Padding import pad, unpad
# Key must be 8 bytes
key = b'8bytekey' # Shared secret key
# Message to send
message = "Meet me very urgently"
# Encrypt at sender (User A)
cipher = DES.new(key, DES.MODE_ECB)
encrypted = cipher.encrypt(pad(message.encode(), DES.block_size))
print("Encrypted:", encrypted)
# Decrypt at receiver (User B)
decipher = DES.new(key, DES.MODE_ECB)
decrypted = unpad(decipher.decrypt(encrypted), DES.block_size)
print("Decrypted:", decrypted.decode())

import hashlib
# Message to hash
message = "Sanjivani University"
# Generate MD5 hash
md5_hash = hashlib.md5(message.encode()).hexdigest()
print("MD5 Hash:", md5_hash)
# Generate SHA-256 hash
sha256_hash = hashlib.sha256(message.encode()).hexdigest()
print("SHA-256 Hash:", sha256_hash)
# Verification example
def verify_hash(input_text, original_hash,
algorithm='sha256'):
if algorithm == 'md5':
new_hash =
hashlib.md5(input_text.encode()).hexdigest()
elif algorithm == 'sha256':
new_hash =
hashlib.sha256(input_text.encode()).hexdigest()
else:
return False
return new_hash == original_hash
# Verify both hashes
print("MD5 Verified:", verify_hash("Sanjivani University",
md5_hash, 'md5'))
print("SHA-256 Verified:", verify_hash("Sanjivani
University", sha256_hash, 'sha256'))

14. Student Class Example TypeScript

class Student {
  name: string;
  rollNo: number;
  marks: number;

  constructor(name: string, rollNo: number, marks: number) {
    this.name = name;
    this.rollNo = rollNo;
    this.marks = marks;
  }

  calculateGrade(): string {
    if(this.marks >= 90) return 'A+';
    else if(this.marks >= 80) return 'A';
    else if(this.marks >= 70) return 'B+';
    else if(this.marks >= 60) return 'B';
    else if(this.marks >= 50) return 'C';
    else return 'F';
  }

  display(): void {
    console.log(`Name: ${this.name}`);
    console.log(`Roll No: ${this.rollNo}`);
    console.log(`Marks: ${this.marks}`);
    console.log(`Grade: ${this.calculateGrade()}`);
  }
}

const student1 = new Student('Yuvraj', 101, 85);
student1.display();

const student2 = new Student('Anjali', 102, 72);
student2.display();

    const students = [];

    function addStudent() {
      const name = document.getElementById('studentName').value.trim();
      const marks = parseFloat(document.getElementById('studentMarks').value);
      if(!name || isNaN(marks)) return;

      students.push({name, marks});
      displayTable();
      document.getElementById('studentName').value = '';
      document.getElementById('studentMarks').value = '';
    }

    function displayTable() {
      const tbody = document.querySelector('#marksTable tbody');
      tbody.innerHTML = '';
      let total = 0;

      students.forEach(student => {
        const row = document.createElement('tr');
        row.innerHTML = `<td>${student.name}</td><td>${student.marks}</td>`;
        tbody.appendChild(row);
        total += student.marks;
      });

      const avg = students.length > 0 ? (total / students.length).toFixed(2) : 0;
      document.getElementById('summary').textContent = `Total: ${total} | Average: ${avg}`;
    }
  </script>
</body>

// Simulated Angular Component Logic

const courses = [
  { id: 'C101', name: 'Web Development', duration: '3 months' },
  { id: 'C102', name: 'Data Science', duration: '4 months' },
  { id: 'C103', name: 'Machine Learning', duration: '5 months' },
  { id: 'C104', name: 'Cybersecurity', duration: '3 months' }
];

const tbody = document.querySelector('#courseTable tbody');

function displayCourses(courseList) {
  tbody.innerHTML = '';
  courseList.forEach(course => {
    const row = document.createElement('tr');
    row.innerHTML = `
      <td>${course.id}</td>
      <td>${course.name}</td>
      <td>${course.duration}</td>
    `;
    tbody.appendChild(row);
  });
}

// Initialize table
displayCourses(courses);
