def caesar_encrypt(text, shift):
result = ""
for char in text:
if char.isalpha():
base = ord('A') if char.isupper() else ord('a')
result += chr((ord(char) - base + shift) % 26 + base)
else:
result += char
return result
def caesar_decrypt(cipher, shift):
return caesar_encrypt(cipher, -shift)

3. Hill Cipher encryption and decryption Python

<
import numpy as np
# Convert text to numbers (A=0, B=1, ..., Z=25)
def text_to_numbers(text):
return [ord(char.upper()) - ord('A') for char in text if char.isalpha()]
# Convert numbers back to text
def numbers_to_text(numbers):
return ''.join([chr(num % 26 + ord('A')) for num in numbers])
# Pad text to match matrix size
def pad_text(text, size):
while len(text) % size != 0:
text += 'X' # Padding with 'X'
return text
# Encryption: C = (K × P) mod 26
def hill_encrypt(plain_text, key_matrix):
size = key_matrix.shape[0]
plain_text = pad_text(plain_text, size)
numbers = text_to_numbers(plain_text)
cipher = []
for i in range(0, len(numbers), size):
block = np.array(numbers[i:i+size])
encrypted_block = np.dot(key_matrix, block) % 26
cipher.extend(encrypted_block)
return numbers_to_text(cipher)
# Decryption: P = (K⁻¹ × C) mod 26
def hill_decrypt(cipher_text, key_matrix):
size = key_matrix.shape[0]
numbers = text_to_numbers(cipher_text)
# Compute inverse of key matrix mod 26
det = int(np.round(np.linalg.det(key_matrix)))
det_inv = pow(det, -1, 26) # Modular inverse of determinant
adj = np.round(det * np.linalg.inv(key_matrix)).astype(int) % 26
inverse_matrix = (det_inv * adj) % 26
plain = []
for i in range(0, len(numbers), size):
block = np.array(numbers[i:i+size])
decrypted_block = np.dot(inverse_matrix, block) % 26
plain.extend(decrypted_block)
return numbers_to_text(plain)
# Example usage
key = np.array([[3, 3],
[2, 5]]) # Must be invertible mod 26
text = "HELLO"
encrypted = hill_encrypt(text, key)
decrypted = hill_decrypt(encrypted, key)
print("Original Text:", text)
print("Encrypted Text:", encrypted)
print("Decrypted Text:", decrypted)

from Crypto.Cipher import DES
from Crypto.Util.Padding import pad, unpad
# Key must be 8 bytes
key = b'8bytekey' # Shared secret key
# Message to send
message = "Meet me very urgently"
# Encrypt at sender (User A)
cipher = DES.new(key, DES.MODE_ECB)
encrypted = cipher.encrypt(pad(message.encode(), DES.block_size))
print("Encrypted:", encrypted)
# Decrypt at receiver (User B)
decipher = DES.new(key, DES.MODE_ECB)

def verify_hash(input_text, original_hash,

if algorithm == 'md5':
new_hash =
hashlib.md5(input_text.encode()).hexdigest()

new_hash =

13. Q3. Write a program to perform encryption and decryption using Hill cipher simpel code Python

<def to_num(text): return [ord(c)-65 for c in text.upper()]
def to_text(nums): return ''.join([chr(n+65) for n in nums])
def encrypt(msg, k):
 if len(msg)%2: msg += 'X'
 nums = to_num(msg)
 res = []
 for i in range(0,len(nums),2):
 a,b = nums[i], nums[i+1]
 res += [(k[0][0]*a + k[0][1]*b)%26, (k[1][0]*a + k[1][1]*b)%26]
 return to_text(res)
def decrypt(msg, k):
 det = k[0][0]*k[1][1]-k[0][1]*k[1][0]
 inv = pow(det, -1, 26)
 k_inv = [[k[1][1]*inv%26, -k[0][1]*inv%26], [-k[1][0]*inv%26, k[0][0]*inv%26]]
 return encrypt(msg, k_inv) # reuse encrypt with inverse key
# Example
key = [[3,3],[2,5]]
msg = "HELP"
cipher = encrypt(msg,key)
print("Encrypted:", cipher)
print("Decrypted:", decrypt(cipher,key))
                
'''Encryption part
1. Import the required libraries: DES for encryption/decryption and pad, unpad for data padding.
2. Define an 8-byte secret key → key = b'8bytekey'.
3. Enter the plaintext message to encrypt → "Meet me very urgently".
4. Create a DES cipher object using the key and ECB mode → DES.new(key, DES.MODE_ECB).
5. Convert the message to bytes and pad it to make its length a multiple of 8 bytes.
6. Encrypt the padded message using the cipher object → produces ciphertext.
7. Print the encrypted message (ciphertext).
Decryption part
