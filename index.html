Find minimum of provided set of numbers

#!/bin/bash
echo "Enter numbers separated by space:"
read -a nums
min=${nums[0]}
for i in "${nums[@]}"; do
    if [ $i -lt $min ]; then min=$i; fi
done
echo "Minimum number is: $min"
chmod +x min.sh && ./min.sh

Determine whether a given year is leap year

#!/bin/bash
echo -n "Enter year: "
read y
if [ $((y % 400)) -eq 0 ] || ([ $((y % 4)) -eq 0 ] && [ $((y % 100)) -ne 0 ]); then
    echo "$y is a Leap Year"
else
    echo "$y is not a Leap Year"
fi
chmod +x leap.sh && ./leap.sh

Determine whether a string is palindrome

#!/bin/bash
echo -n "Enter string: "
read str
rev=$(echo $str | rev)
if [ "$str" == "$rev" ]; then
    echo "String is Palindrome"
else
    echo "String is not Palindrome"
fi
chmod +x pal.sh && ./pal.sh

Determine whether a given number is prime

#!/bin/bash
echo -n "Enter number: "
read n
for((i=2; i<=n/2; i++)); do
  if [ $((n%i)) -eq 0 ]; then
    echo "$n is not prime"
    exit
  fi
done
echo "$n is prime"
chmod +x prime.sh && ./prime.sh

Find sum of digits of a number

#!/bin/bash
echo -n "Enter number: "
read n
sum=0
while [ $n -gt 0 ]; do
    sum=$((sum + n % 10))
    n=$((n / 10))
done
echo "Sum of digits: $sum"
chmod +x sum_digits.sh && ./sum_digits.sh

Determine average of provided set of numbers

#!/bin/bash
echo "Enter numbers separated by space:"
read -a nums
sum=0
count=${#nums[@]}
for i in "${nums[@]}"; do
    sum=$((sum + i))
done
avg=$(echo "scale=2; $sum / $count" | bc)
echo "Average: $avg"
chmod +x avg.sh && ./avg.sh
