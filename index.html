def caesar_encrypt(text, shift):
result = ""
for char in text:
if char.isalpha():
base = ord('A') if char.isupper() else ord('a')
result += chr((ord(char) - base + shift) % 26 + base)
else:
result += char
return result
def caesar_decrypt(cipher, shift):
return caesar_encrypt(cipher, -shift)

      const now = new Date();
      let hours = now.getHours();
      let minutes = now.getMinutes();
      let seconds = now.getSeconds();

      hours = hours < 10 ? '0'+hours : hours;
      minutes = minutes < 10 ? '0'+minutes : minutes;
      seconds = seconds < 10 ? '0'+seconds : seconds;

      document.getElementById('clock').textContent = `${hours}:${minutes}:${seconds}`;

3. Employee Salary Calculator TypeScript

class Employee {
  name: string;
  basicPay: number;
  hra: number;
  deductions: number;

  constructor(name: string, basicPay: number, hra: number, deductions: number) {
    this.name = name;
    this.basicPay = basicPay;
    this.hra = hra;
    this.deductions = deductions;
  }

  calculateNetSalary(): number {
    return this.basicPay + this.hra - this.deductions;
  }

  displaySalary(): void {
    console.log(`Employee Name: ${this.name}`);
    console.log(`Basic Pay: ₹${this.basicPay}`);
    console.log(`HRA: ₹${this.hra}`);
    console.log(`Deductions: ₹${this.deductions}`);
    console.log(`Net Salary: ₹${this.calculateNetSalary()}`);
  }
}

const emp1 = new Employee('Yuvraj', 50000, 12000, 8000);
emp1.displaySalary();

const emp2 = new Employee('Anjali', 40000, 10000, 5000);
emp2.displaySalary();

  <div class="quiz-container">
    <h2 id="question">Question text</h2>
    <ul class="options" id="options"></ul>
    <button class="btn" id="nextBtn">Next</button>
    <div class="score" id="score"></div>
  </div>

  <script>
    const quizData = [
      { question: "What is the capital of France?", options: ["Paris", "London", "Berlin", "Rome"], answer: "Paris" },
      { question: "2 + 2 = ?", options: ["3","4","5","6"], answer: "4" },
      { question: "Which language is used for web apps?", options: ["Python","JavaScript","C++","Java"], answer: "JavaScript" }
    ];

    let currentQuestion = 0, score = 0;

    function loadQuestion() {
      const q = quizData[currentQuestion];
      document.getElementById('question').textContent = q.question;
      const optionsEl = document.getElementById('options');
      optionsEl.innerHTML = '';
      q.options.forEach(option => {
        const li = document.createElement('li');
        li.textContent = option;
        li.onclick = function() {
          if(option === q.answer) score++;
          document.getElementById('nextBtn').style.display = 'inline-block';
        };
        optionsEl.appendChild(li);
      });
    }

    document.getElementById('nextBtn').onclick = function() {
      currentQuestion++;
      if(currentQuestion < quizData.length){
        loadQuestion();
        this.style.display = 'none';
      } else {
        document.getElementById('question').textContent = 'Quiz Completed!';
        document.getElementById('options').innerHTML = '';
        this.style.display = 'none';
        document.getElementById('score').textContent = `Your Score: ${score} / ${quizData.length}`;
      }
    };

    loadQuestion();
    document.getElementById('nextBtn').style.display = 'none';
  </script>
</body>
</html>

class Student {
  name: string;
  rollNo: number;
  marks: number;

  constructor(name: string, rollNo: number, marks: number) {
    this.name = name;
    this.rollNo = rollNo;
    this.marks = marks;
  }

  calculateGrade(): string {
    if(this.marks >= 90) return 'A+';
    else if(this.marks >= 80) return 'A';
    else if(this.marks >= 70) return 'B+';
    else if(this.marks >= 60) return 'B';
    else if(this.marks >= 50) return 'C';
    else return 'F';
  }

  display(): void {
    console.log(`Name: ${this.name}`);
    console.log(`Roll No: ${this.rollNo}`);
    console.log(`Marks: ${this.marks}`);
    console.log(`Grade: ${this.calculateGrade()}`);
  }
}

student1.display();

const student2 = new Student('Anjali', 102, 72);

    const students = [];

    function addStudent() {
      const name = document.getElementById('studentName').value.trim();
      const marks = parseFloat(document.getElementById('studentMarks').value);
      if(!name || isNaN(marks)) return;

      students.push({name, marks});
      displayTable();
      document.getElementById('studentName').value = '';
      document.getElementById('studentMarks').value = '';

      const tbody = document.querySelector('#marksTable tbody');
      tbody.innerHTML = '';

        row.innerHTML = `<td>${student.name}</td><td>${student.marks}</td>`;
        tbody.appendChild(row);
        total += student.marks;
      });

      const avg = students.length > 0 ? (total / students.length).toFixed(2) : 0;

    }
