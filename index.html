Shell Script to Determine whether a Number is Odd or Even

#!/bin/bash
echo -n "Enter a number: "
read n

if [ $((n % 2)) -eq 0 ]; then
    echo "$n is Even"
else
    echo "$n is Odd"
fi

Shell Script to Determine whether a Year is Leap Year or not

#!/bin/bash
echo -n "Enter year (YYYY): "
read y

if [ $((y % 400)) -eq 0 ]; then
    echo "$y is a Leap Year"
elif [ $((y % 100)) -ne 0 ] && [ $((y % 4)) -eq 0 ]; then
    echo "$y is a Leap Year"
else
    echo "$y is not a Leap Year"
fi

Commands to run:

chmod +x leap.sh && ./leap.sh

Shell Script to Determine whether a Number is Prime or not

#!/bin/bash
echo -n "Enter a number: "
read n

if [ $n -lt 2 ]; then
    echo "$n is not prime"
    exit
fi

for ((i=2; i<=n/2; i++))
do
    if [ $((n % i)) -eq 0 ]; then
        echo "$n is not a prime number"
        exit
    fi
done

echo "$n is a prime number"

Commands to run:

chmod +x prime.sh && ./prime.sh

Shell Script to Determine if a Triplet follows Pythagoras Theorem

#!/bin/bash
echo "Enter side a: "; read a
echo "Enter side b: "; read b
echo "Enter hypotenuse c: "; read c

lhs=$((a*a + b*b))
rhs=$((c*c))

if [ $lhs -eq $rhs ]; then
    echo "This is a Pythagorean Triplet."
else
    echo "This is NOT a Pythagorean Triplet."
fi

Commands to run:

chmod +x pythagoras.sh && ./pythagoras.sh

Shell Script to Determine Maximum and Minimum of three Numbers

#!/bin/bash
echo "Enter three numbers separated by spaces: "
read a b c

max=$a
if [ $b -gt $max ]; then max=$b; fi
if [ $c -gt $max ]; then max=$c; fi

min=$a
if [ $b -lt $min ]; then min=$b; fi
if [ $c -lt $min ]; then min=$c; fi

echo "Maximum number is: $max"
echo "Minimum number is: $min"

Commands to run:

chmod +x maxmin.sh && ./maxmin.sh
