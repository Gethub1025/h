1. Hospital Patient Management System

import java.util.*;
class Patient {
    int id; String name, disease, doctor;
    Patient(int id,String name,String disease,String doctor){
        this.id=id; this.name=name; this.disease=disease; this.doctor=doctor;
    }
}
public class HospitalSystem {
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        ArrayList<Patient> list=new ArrayList<>();
        while(true){
            System.out.println("1.Add 2.Display 3.Search 4.Delete 5.Count 6.Exit");
            int ch=sc.nextInt();
            switch(ch){
                case 1:
                    list.add(new Patient(sc.nextInt(),sc.next(),sc.next(),sc.next()));
                    break;
                case 2:
                    for(Patient p:list) System.out.println(p.id+" "+p.name);
                    break;
                case 3:
                    int id=sc.nextInt();
                    for(Patient p:list) if(p.id==id) System.out.println("Found");
                    break;
                case 4:
                    id=sc.nextInt();
                    list.removeIf(p->p.id==id);
                    break;
                case 5:
                    System.out.println(list.size());
                    break;
                case 6: return;
            }
        }
    }
}

2. Electricity Bill Calculation

import java.util.*;
class Bill{
    int id,units; String name;
    Bill(int id,String name,int units){
        this.id=id; this.name=name; this.units=units;
    }
    double calc(){
        if(units<=100) return units*5;
        else if(units<=200) return 100*5+(units-100)*7;
        else return 100*5+100*7+(units-200)*10;
    }
}
public class ElectricityBill{
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        ArrayList<Bill> list=new ArrayList<>();
        int n=sc.nextInt();
        for(int i=0;i<n;i++){
            list.add(new Bill(sc.nextInt(),sc.next(),sc.nextInt()));
        }
        for(Bill b:list)
            System.out.println(b.name+" ₹"+b.calc());
    }
}

3. Inventory Management System

import java.util.*;

    int id,qty; String name; double price;
    Product(int id,String name,double price,int qty){
        this.id=id; this.name=name; this.price=price; this.qty=qty;
    }
}
public class Inventory{
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        ArrayList<Product> list=new ArrayList<>();
        while(true){
            int ch=sc.nextInt();
            switch(ch){
                case 1:
                    list.add(new Product(sc.nextInt(),sc.next(),sc.nextDouble(),sc.nextInt()));
                    break;
                case 2:
                    for(Product p:list) System.out.println(p.name+" "+p.qty);
                    break;
                case 3:
                    int id=sc.nextInt(),q=sc.nextInt();
                    for(Product p:list) if(p.id==id) p.qty=q;
                    break;
                case 4:
                    double total=0;

                    System.out.println(total);

                    id=sc.nextInt();
                    for(Product p:list) if(p.id==id) System.out.println("Found");
                    break;
                case 6:return;
            }
        }
    }
}

4. Online Voting System

import java.util.*;
public class VotingSystem{
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        int[] votes=new int[3];
        HashSet<Integer> set=new HashSet<>();
        while(true){
            int id=sc.nextInt();
            if(id==0) break;
            if(set.contains(id)){
                System.out.println("Already voted");
                continue;
            }
            int c=sc.nextInt();
            if(c>=1&&c<=3){
                votes[c-1]++;
                set.add(id);
            }
        }
        int max=0,w=0;
        for(int i=0;i<3;i++){
            if(votes[i]>max){max=votes[i]; w=i;}
        }
        System.out.println("Winner: "+(w+1));
    }
}
