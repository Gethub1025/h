2. Electricity Bill Calculation

import java.util.Scanner;

public class ElectricityBill {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        int[] id = new int[n];
        String[] name = new String[n];
        int[] units = new int[n];

        for (int i = 0; i < n; i++) {
            id[i]=sc.nextInt();
            name[i]=sc.next();
            units[i]=sc.nextInt();
        }

        for (int i = 0; i < n; i++) {
            double bill=0;
            if(units[i]<=100) bill=units[i]*5;
            else if(units[i]<=200) bill=100*5+(units[i]-100)*7;
            else bill=100*5+100*7+(units[i]-200)*10;

            System.out.println(name[i]+" Rs."+bill);
        }
    }
}
